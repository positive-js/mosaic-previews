onPositionChange: ($event: ConnectedOverlayPositionChange): void => { if (!this.instance) { return; } let newPlacement = this.placement; const { originX, originY, overlayX, overlayY } = $event.connectionPair; Object.keys(this.availablePositions).some((key) => { if (originX === this.availablePositions[key].originX && originY === this.availablePositions[key].originY && overlayX === this.availablePositions[key].overlayX && overlayY === this.availablePositions[key].overlayY) { newPlacement = key as PopUpPlacements; return true; } return false; }); this.placementChange.emit(newPlacement); this.updateClassMap(newPlacement); if ($event.scrollableViewProperties.isOverlayClipped && this.instance.isVisible()) { this.ngZone.run(() => this.hide()); } }

'' } } }, en: { relative: { long: { beforeYesterdayNotCurrentYear: '', beforeYesterdayCurrentYear: '', beforeYesterdayCurrentYearSeconds: '', beforeYesterdayCurrentYearMilliseconds: '', yesterday: '',
